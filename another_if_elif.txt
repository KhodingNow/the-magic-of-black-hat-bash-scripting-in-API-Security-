#!/bin/bash

CUSTOMER_INPUT="${1}"

if [[ -z "${CUSTOMER_INPUT}" ]]; then
    echo "You must provide an arg"
    exit 1
fi

if [[ -f "${CUSTOMER_INPUT}" ]]; then
    echo "${CUSTOMER_INPUT} is a file"

elif [[ -d "${CUSTOMER_INPUT}" ]]; then
    echo "${CUSTOMER_INPUT} is a directory" 

else

    echo "${CUSTOMER_INPUT} is not a file or directory"
